# 配列まるごと渡せば OK!

## 子

```html
// LinkBase.vue (子)

<template lang="pug">
    v-card
      nuxt-link(:to="item.url")
        p {{ item.textEnglish }}
        p {{ item.textJapanese }}
</template>

<script>
  export default {
    props: {
      item: Object,
    },
  };
</script>
```

## 親

```html
// Links (親)

<template lang="pug">
          v-row
            span(
              v-for="(item, i) in items"
              :key="i"
              )
              BaseLink(:item="item")
</template>

<script>
  import BaseLink from "@/components/BaseLink.vue";

  export default {
    components: {
      BaseLink,
    },
    data: () => ({
      items: [
        {
          textEnglish: "WHO WE ARE",
          textJapanese: "私たちについて",
          url: "/about",
        },
        {
          textEnglish: "WHAT WE DO",
          textJapanese: "サービス・製品",
          url: "/service",
        },
        {
          textEnglish: "NEWS",
          textJapanese: "ニュース",
          url: "/news",
        },
      ],
    }),
  };
</script>
```
